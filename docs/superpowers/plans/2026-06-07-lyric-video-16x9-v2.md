# Lyric Video 16:9 v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade the current render-worker sample into an official-music-work-feeling 16:9 lyric MV that outputs playable MP4 with both video and audio tracks.

**Architecture:** Keep Remotion as the production render engine. Java `LocalProcessVideoRenderService` stages object-storage audio and cover assets into a render temp directory, passes concrete local file paths to Node, and stores the rendered MP4/timeline back to object storage. The Node render-worker renders `LyricVideo16x9V2`, burns sentence-level lyrics into safe areas, includes audio via Remotion `<Audio>`, and validates output with ffprobe before reporting success.

**Tech Stack:** Java 21, Spring Boot module boundaries, local/S3-compatible object storage, Node.js 22, TypeScript, Remotion 4, FFmpeg/FFprobe, Gradle, Node test runner.

public byte[] getObject(String objectKey) {
  String safeObjectKey = ObjectStorageKeys.requireSafeObjectKey(objectKey);

  if (!target.startsWith(rootDirectory)) {

}

private Path stageObject(Path tempDirectory, String objectKey, String fileName) {
  try {
    Path assetDirectory = tempDirectory.resolve("input-assets");
    Files.createDirectories(assetDirectory);
    Path target = assetDirectory.resolve(fileName).normalize();
    if (!target.startsWith(assetDirectory.normalize())) {
      throw new IllegalStateException("Render input asset path escapes temp directory");

}

import {AbsoluteFill, Audio, Img, interpolate, useCurrentFrame, useVideoConfig} from 'remotion';
import type {LyricLine, LyricVideoProps} from './render-input';
import {videoColors, videoTypography} from './video-style';

function activeLine(lines: readonly LyricLine[], frame: number): LyricLine {
  return (
    lines.find((line) => frame >= line.startFrame && frame <= line.endFrame) ??
    lines[lines.length - 1] ?? {startFrame: 0, endFrame: 1, text: ''}
  );
}

export function LyricVideo16x9V2(props: LyricVideoProps) {
  const frame = useCurrentFrame();
  const {durationInFrames} = useVideoConfig();
  const line = activeLine(props.lyrics, frame);
  const introOpacity = interpolate(frame, [0, 24, 72], [0, 1, 1], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });
  const progress = Math.min(1, frame / Math.max(1, durationInFrames - 1));
  const backgroundScale = interpolate(frame, [0, durationInFrames], [1.1, 1.15], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  return (
    <AbsoluteFill style={{backgroundColor: videoColors.ink, color: videoColors.ivory, fontFamily: '-apple-system, BlinkMacSystemFont, "Segoe UI", sans-serif'}}>
      <Audio src={props.audioSource} />
      <Img src={props.coverSource} style={{position: 'absolute', inset: -60, width: 2040, height: 1200, objectFit: 'cover', filter: 'blur(24px)', opacity: 0.52, transform: `scale(${backgroundScale})`}} />
      <AbsoluteFill style={{background: 'linear-gradient(90deg, rgba(11,17,16,0.9), rgba(11,17,16,0.42) 48%, rgba(11,17,16,0.78))'}} />
      <div style={{position: 'absolute', left: 120, top: 96, color: videoColors.gold, fontSize: videoTypography.meta, letterSpacing: 0}}>燕云乐坊</div>
      <Img src={props.coverSource} style={{position: 'absolute', left: 120, top: 214, width: 590, height: 590, objectFit: 'cover', boxShadow: '0 42px 120px rgba(0,0,0,0.48)', border: '1px solid rgba(246,237,220,0.18)'}} />
      <div style={{position: 'absolute', left: 770, top: 238, width: 880, opacity: introOpacity}}>
        <div style={{fontSize: videoTypography.title, fontWeight: 750, lineHeight: 1.04}}>{props.songTitle}</div>
        <div style={{marginTop: 24, maxWidth: 760, color: videoColors.mutedIvory, fontSize: videoTypography.subtitle, lineHeight: 1.4}}>{props.songSummary}</div>
      </div>
      <div style={{position: 'absolute', left: 770, right: 120, bottom: 210, minHeight: 156, fontSize: videoTypography.lyric, fontWeight: 680, lineHeight: 1.24, textShadow: '0 4px 22px rgba(0,0,0,0.75)'}}>{line.text}</div>
      <div style={{position: 'absolute', left: 770, right: 120, bottom: 128, height: 48, display: 'flex', alignItems: 'end', gap: 5, opacity: 0.78}}>
        {Array.from({length: 42}).map((_, index) => {
          const height = 10 + ((index * 17 + frame * 2) % 34);
          return <div key={index} style={{width: 7, height, backgroundColor: videoColors.gold, opacity: 0.34 + (index % 5) * 0.08}} />;
        })}
      </div>
      <div style={{position: 'absolute', left: 120, right: 120, bottom: 78, height: 3, backgroundColor: 'rgba(246,237,220,0.18)'}}>
        <div style={{height: 3, width: `${progress * 100}%`, backgroundColor: videoColors.gold}} />
      </div>
    </AbsoluteFill>
  );
}
```

import {execFile} from 'node:child_process';
import {promisify} from 'node:util';

const execFileAsync = promisify(execFile);

async function assertPlayableMp4(videoFilePath: string): Promise<void> {
  const {stdout} = await execFileAsync('ffprobe', [
    '-v',
    'error',
    '-show_entries',
    'stream=codec_type,codec_name',
    '-of',
    'json',
    videoFilePath,
  ]);
  const parsed = JSON.parse(stdout) as {streams?: Array<{codec_type?: string; codec_name?: string}>};
  const streams = parsed.streams ?? [];
  if (!streams.some((stream) => stream.codec_type === 'video')) {
    throw new Error('Rendered MP4 is missing video stream');
  }
  if (!streams.some((stream) => stream.codec_type === 'audio')) {
    throw new Error('Rendered MP4 is missing audio stream');
  }
}
```
